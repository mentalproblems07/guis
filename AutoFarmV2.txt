-- Makes a toggle, prints "On!" while switched on, "Off!" while turned off.
lib:AddToggle("Anti-Admin", function(state)
	if state then
		getgenv().antiMod = state
		if getgenv().antiMod then
			local rolesToKick = {"ðŸ’Žï½œSpecial Member", "âš¡ï½œModerator", "âœ¨ï½œAdmin", "ðŸ‘¾ï½œDev", "ðŸ‘‘ï½œCreator"}

			local function onPlayer(player)

					return player:GetRoleInGroup(15762035)
				end)

				if success and (table.find(rolesToKick, role) or player:IsInGroup(16817874)) then
					Players.LocalPlayer:Kick(player.Name .. " joined.")
				end
			end

			game.Players.PlayerAdded:Connect(onPlayer)
			for _, player in ipairs(Players:GetPlayers()) do
				if player ~= Players.LocalPlayer then
					onPlayer(player)
				end
			end

	end
end, false) -- 'true' or 'false', to make it default, if it's 'true' it will toggled on by default and vice versa.




lib:AddToggle("Farm FS", function(state)
	if state then
		getgenv().farming = true
		spawn(function()
			while getgenv().farming do
				task.wait(0.1)
				local args = {
					[1] = "Add_FS_Request"
				}
				game.ReplicatedStorage.RemoteEvent:FireServer(args)
			end
		end)
	else
		getgenv().farming = false
		print('Off!')
	end
end, false)


lib:AddToggle("Farm Speed", function(state)
	if state then
		getgenv().farmspeed = true
		spawn(function()
			while getgenv().farmspeed do
				task.wait(0.1)		
				game.ReplicatedStorage:WaitForChild("RemoteEvent"):FireServer({[1] = "Add_MS_Request"})
			end
		end)
	else
		getgenv().farmspeed = false
		print('Off!')
	end
end, false)

lib:AddToggle("Farm Jump", function(state)
	if state then
		getgenv().farmjump = true
		spawn(function()
			while getgenv().farmjump do					
				game.ReplicatedStorage:WaitForChild("RemoteEvent"):FireServer({[1] = "Add_JF_Request"})
				task.wait(0.1)
			end
		end)	
	else
		getgenv().farmjump = false
		print('Off!')
	end
end, false)
